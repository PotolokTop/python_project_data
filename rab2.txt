class Node:
    def __init__(self, v):
        self.v = v
        self.l = None
        self.r = None

root = Node(2)
root.l = Node(3)
root.r = Node(5)

def f(n):
    if n.l is None and n.r is None:
        return n.v
    return (f(n.l) if n.l else 1) * (f(n.r) if n.r else 1)

print(f(root))
